Description:
  • Retrieve autocomplete results from Constructor.io API
Source:
  • modules/autocomplete.js, line 134
See:
  • https://docs.constructor.com/reference/autocomplete-autocomplete-results
Example
constructorio.autocomplete.getAutocompleteResults('t-shirt', {
    resultsPerSection: {
        Products: 5,
        'Search Suggestions': 10,
    },
    filters: {
        size: 'medium'
    },
});
